 10 /**
 11  * Class Cell
 12  *
 13  * @constructor
 14  */
 15 function Cell() {
 16     this.type = null; // plains, mountain, swamp
 17     this.x = -1;
 18     this.y = -1;
 19 }
 20 
 21 Cell.prototype = {
 22 
 23     /**
 24      * Transform the cell's attributs to a JSON string
 25      * @return data, contains the unit attributs.
 26      */
 27     toJSON : function() {
 28 
 29         var data = {
 30             "type": this.type,
 31             "x" : this.x ,
 32             "y" : this.y,
 33         };
 34 
 35         return data;
 36     },
 37 }
 38 
 39 module.exports = Cell;
